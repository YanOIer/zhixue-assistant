# 后端开发指南 - FastAPI + RAG

## 基本信息
- **开发平台**：Python 3.9+
- **Web框架**：FastAPI
- **数据库**：SQLite（开发）/ PostgreSQL（生产）
- **AI框架**：SentenceTransformers + FAISS + DeepSeek API
- **部署**：Uvicorn + Gunicorn

```python
# 依赖：获取当前用户
async def get_current_user(token: str = Header(...)):
    # JWT验证逻辑
    return verify_token(token)

# ========== 文档API ==========

@app.post("/api/documents/upload")
async def upload_document(
    file: UploadFile = File(...),
    user: dict = Depends(get_current_user)
):
    """上传并处理文档"""
    # 生成唯一ID
    doc_id = str(uuid.uuid4())
    
    # 保存文件
    file_path = f"uploads/{doc_id}_{file.filename}"
    with open(file_path, "wb") as f:
        content = await file.read()
        f.write(content)
    
    # 异步处理文档（提取文本、建立索引）
    process_document.delay(doc_id, file_path)
    
    return {
        "id": doc_id,
        "name": file.filename,
        "status": "processing",
        "message": "文档上传成功，正在处理中"
    }

@app.get("/api/documents")
async def get_documents(user: dict = Depends(get_current_user)):
    """获取用户的文档列表"""
    documents = await db.get_user_documents(user["id"])
    return {"documents": documents}

@app.delete("/api/documents/{doc_id}")

    doc_id: str,

# ========== 对话API ==========

@app.post("/api/chat")
async def chat(
    request: ChatRequest,
    user: dict = Depends(get_current_user)
):
    """流式对话接口"""
    
    async def generate():
        # 1. 检索相关文档
        yield f'data: {{"type": "thinking", "content": "正在检索相关资料..."}}\n\n'
        
        sources = rag.search(request.question, request.document_id)
        
        # 2. 返回参考来源
        yield f'data: {{"type": "source", "content": {json.dumps(sources)}}}\n\n'
        
        # 3. 生成AI回答（流式）
        async for chunk in rag.generate_answer(request.question, sources):
            yield f'data: {{"type": "answer", "content": "{chunk}"}}\n\n'
        
        # 4. 完成
        yield f'data: {{"type": "done"}}\n\n'
    
    return StreamingResponse(
        generate(),
        media_type="text/event-stream"
    )
```

### 2. RAG系统集成 (`ai_module/integrate_with_backend.py`)

```python
"""
RAG系统与后端的集成接口
"""
from typing import List, Dict, AsyncGenerator
from rag_system import RAGSystem
import os

class RAGService:
    """RAG服务单例"""
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._initialized = False
        return cls._instance
    
    def __init__(self):
        if self._initialized:
            return
        
        # 初始化RAG系统
        self.rag = RAGSystem(
            embedding_model='BAAI/bge-small-zh',
            reranker_model='BAAI/bge-reranker-v2-m3',
            use_hnsw=True,
            use_reranker=True,
            device='cpu'  # 根据实际情况调整
        )
        
        # 加载已有索引（如果有）
        index_path = "data/knowledge_base"
        if os.path.exists(f"{index_path}.faiss"):
            self.rag.load(index_path)
        
        self._initialized = True
    
    def add_document(self, text: str, doc_id: str) -> bool:
        """添加文档到知识库"""
        try:
            self.rag.add_document(text, doc_id)
            
            # 保存索引
            self.rag.save("data/knowledge_base")
            return True
        except Exception as e:
            print(f"添加文档失败: {e}")
            return False
    
    def search(self, question: str, doc_id: str = None) -> List[Dict]:
        """检索相关文档"""
        results = self.rag.search(question, top_k=5)
        
        # 如果指定了doc_id，过滤结果
        if doc_id:
            results = [r for r in results if r["doc_id"] == doc_id]
        
        return results
    
    def query(self, question: str, doc_id: str = None) -> Dict:
        """完整查询（检索+回答）"""
        return self.rag.query(question, top_k=5)


# 全局RAG服务实例
rag_service = RAGService()
```

### 3. 文档处理 (`ai_module/document_processor.py`)

```python
"""
文档处理模块 - 提取各种格式文档的文本内容
"""
import os
from typing import Optional

class DocumentProcessor:
    """文档处理器"""
    
    SUPPORTED_TYPES = {
        '.txt': 'text',
        '.md': 'markdown',
        '.pdf': 'pdf',
        '.docx': 'word',
        '.doc': 'word'
    }
    
    @classmethod
    def process(cls, file_path: str) -> Optional[str]:
        """
        处理文档，提取文本
        
        Args:
            file_path: 文档路径
        
        Returns:
            提取的文本内容，失败返回None
        """
        ext = os.path.splitext(file_path)[1].lower()
        
        if ext not in cls.SUPPORTED_TYPES:
            raise ValueError(f"不支持的文件类型: {ext}")
        
        doc_type = cls.SUPPORTED_TYPES[ext]
        
        if doc_type == 'text':
            return cls._process_text(file_path)
        elif doc_type == 'pdf':
            return cls._process_pdf(file_path)
        elif doc_type == 'word':
            return cls._process_word(file_path)
        
        return None
    
    @staticmethod
    def _process_text(file_path: str) -> str:
        """处理纯文本"""
        with open(file_path, 'r', encoding='utf-8') as f:
            return f.read()
    
    @staticmethod
    def _process_pdf(file_path: str) -> str:
        """处理PDF"""
        try:
            import PyPDF2
            text = []
            with open(file_path, 'rb') as f:
                reader = PyPDF2.PdfReader(f)
                for page in reader.pages:
                    text.append(page.extract_text())
            return '\n'.join(text)
        except Exception as e:
            print(f"PDF处理失败: {e}")
            return None
    
    @staticmethod
    def _process_word(file_path: str) -> str:
        """处理Word"""
        try:
            import docx
            doc = docx.Document(file_path)
            return '\n'.join([para.text for para in doc.paragraphs])
        except Exception as e:
            print(f"Word处理失败: {e}")
            return None


# 使用示例
if __name__ == "__main__":
    text = DocumentProcessor.process("test.pdf")
    print(f"提取文本长度: {len(text)}")
```

## 数据库设计

### SQLite表结构

```sql
-- 用户表
CREATE TABLE users (
    id TEXT PRIMARY KEY,
    openid TEXT UNIQUE NOT NULL,  -- 微信openid
    nickname TEXT,
    avatar_url TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 文档表
CREATE TABLE documents (
    id TEXT PRIMARY KEY,
    user_id TEXT NOT NULL,
    name TEXT NOT NULL,
    type TEXT NOT NULL,  -- pdf/docx/txt
    size INTEGER,        -- 文件大小（字节）
    pages INTEGER,       -- 页数
    file_path TEXT,      -- 存储路径
    status TEXT DEFAULT 'processing',  -- processing/processed/error
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- 对话历史表
CREATE TABLE chat_history (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id TEXT NOT NULL,
    document_id TEXT,
    question TEXT NOT NULL,
    answer TEXT NOT NULL,
    sources TEXT,        -- JSON格式，参考来源
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (document_id) REFERENCES documents(id)
);
```
